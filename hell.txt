-nginx-deployment.yaml 

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:latest
        ports:
        - containerPort: 80


kubectl apply -f nginx-deployment.yaml

kubectl get pods
kubectl expose deployment nginx-deployment --type=LoadBalancer --port=80
kubectl get services
